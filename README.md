## Masuk ke directory yang digunakan (cmd)
- virtualenv FLaskEnv
- pip install flask
- pip install mysql-connector-python
- python -m flask
- set FLASK_APP=app.py
- set FLASK_ENV=development
- FlaskEnv\Scripts\activate
- flask run --debug
